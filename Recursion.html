What is Recursion?

Recursion is when a method calls itself to solve a smaller part of a problem. Make sure to define a base case to avoid infinite loops.

void Recurse(int count)
{
    if (count == 0)
        return;

    Console.WriteLine(count);
    Recurse(count - 1); // Recursive call
}

Example 1 – Recursive File Search (Basic)

Scan a directory and its subdirectories for all files.

using System;
using System.IO;

void ScanDirectory(string path)
{
    try
    {
        foreach (string file in Directory.GetFiles(path))
        {
            Console.WriteLine(file);
        }

        foreach (string dir in Directory.GetDirectories(path))
        {
            ScanDirectory(dir); // Recursive call
        }
    }
    catch (UnauthorizedAccessException)
    {
        Console.WriteLine($"Access denied: {path}");
    }
}

// Usage
ScanDirectory("C:\\MyFolder");

Example 2 – Recursive File Search (Console App)

Command-line program that searches all files in a directory (optional: filter by extension).

using System;
using System.IO;

class Program
{
    static void Main(string[] args)
    {
        if (args.Length == 0)
        {
            Console.WriteLine("Usage: FileSearch <path> [<extension>]");
            return;
        }

        string path = args[0];
        string? ext = args.Length > 1 ? args[1] : null;

        Search(path, ext);
    }

    static void Search(string path, string? extension)
    {
        try
        {
            foreach (var file in Directory.GetFiles(path))
            {
                if (extension == null || file.EndsWith(extension, StringComparison.OrdinalIgnoreCase))
                {
                    Console.WriteLine(file);
                }
            }

            foreach (var dir in Directory.GetDirectories(path))
            {
                Search(dir, extension);
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Error: {ex.Message}");
        }
    }
}

Example 3 – Binary Tree In-Order Traversal

Recursively visit each node in a binary tree (left → root → right).

public class TreeNode
{
    public int Value;
    public TreeNode? Left;
    public TreeNode? Right;

    public TreeNode(int value)
    {
        Value = value;
    }
}

void TraverseInOrder(TreeNode? node)
{
    if (node == null) return;

    TraverseInOrder(node.Left);
    Console.WriteLine(node.Value);
    TraverseInOrder(node.Right);
}

// Example usage
var root = new TreeNode(10)
{
    Left = new TreeNode(5),
    Right = new TreeNode(20)
};

TraverseInOrder(root);
